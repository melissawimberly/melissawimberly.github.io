Arrays

Method Structure Example
.slice

slice(index) => obj or nil

slice(start, length) => new_ary or nil

slice(range) => new_ary or nil

a = [ "a", "b", "c", "d", "e" ]

Index

a[2] + a[0] + a[1] #=> "cab"

Start, length

a[0, 2] #=> [ "a", b", "c" ]

Range

a[1..3] #=> [ "b", "c", "d" ]
.group_by

# => 0 => [0, 3, 6, 9, 12, 15]

String Example

bellas = ["Beca", "Aubrey", "Chloe", "Fat Amy", "Cynthia-Rose", "Lilly"]

bellas.group_by { |singer| singer.length}

# => {4=>["Beca"], 6=>["Aubrey"], 5=>["Chloe", "Lilly"], 7=>["Fat Amy"], 12=>["Cynthia-Rose"]}
